### Get the User Profile

```tsx
import { safeatob } from "@toruslabs/openlogin-utils";

const parseToken = (token: any) => {
  try {
    const token = token.split(".")[1]; // payload
    return JSON.parse(safeatob(token));
  } catch (err) {
    console.error(err);
    return null;
  }
};

const userProfile = parseToken(idToken);
```
